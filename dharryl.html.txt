    <!-- Script Implementations Logic Engine -->
    <script>
        let selectedRoomPrice = 0;

        function toggleMenu() {
            const menu = document.getElementById('mobileMenu');
            menu.classList.toggle('hidden');
        }

        function selectRoom(roomName, price) {
            const roomDropdown = document.getElementById('roomType');
            roomDropdown.value = roomName;
            selectedRoomPrice = price;
            
            calculateNights();
            
            // Auto smooth scroll to reservation panel
            document.getElementById('reservation').scrollIntoView({ behavior: 'smooth' });
        }

        function updatePriceFromDropdown() {
            const dropdown = document.getElementById('roomType');
            const selectedOption = dropdown.options[dropdown.selectedIndex];
            selectedRoomPrice = parseFloat(selectedOption.getAttribute('data-price')) || 0;
            calculateNights();
        }

        function calculateNights() {
            const checkInVal = document.getElementById('checkIn').value;
            const checkOutVal = document.getElementById('checkOut').value;
            
            if (!checkInVal || !checkOutVal) return;

            const d1 = new Date(checkInVal);
            const d2 = new Date(checkOutVal);
            
            // Enforce sequential chronology 
            if (d2 <= d1) {
                alert("Check-out date must be strictly after the check-in date.");
                document.getElementById('checkOut').value = "";
                return;
            }

            const timeDiff = Math.abs(d2.getTime() - d1.getTime());
            const totalNights = Math.ceil(timeDiff / (1000 * 60 * 60 * 24));
            
            document.getElementById('daysDisplay').innerText = `${totalNights} Night(s)`;
            
            const totalCost = totalNights * selectedRoomPrice;
            document.getElementById('priceDisplay').innerText = `₱${totalCost.toLocaleString('en-US', {minimumFractionDigits: 2})}`;
        }

        function handleFormSubmit(e) {
            e.preventDefault();
            
            // Destructure variables from standard elements
            const name = document.getElementById('guestName').value;
            const room = document.getElementById('roomType').value;
            const checkIn = document.getElementById('checkIn').value;
            const checkOut = document.getElementById('checkOut').value;
            const totalCostText = document.getElementById('priceDisplay').innerText;
            const totalNightsText = document.getElementById('daysDisplay').innerText;

            if(!room) {
                alert("Please select a valid accommodation room type tier.");
                return;
            }

            // Fire custom dynamic PDF Builder Engine via local Client Canvas
            const { jsPDF } = window.jspdf;
            const doc = new jsPDF();

            // Structure Elegant Styling Header layout assets
            doc.setFillColor(212, 175, 55); // Premium Gold Brand color
            doc.rect(0, 0, 210, 40, 'F');

            doc.setTextColor(255, 255, 255);
            doc.setFont("playfair", "bold");
            doc.setFontSize(24);
            doc.text("DHARRYL HOTEL VOUCHER", 20, 26);

            // Invoice Text Details Metas Block
            doc.setTextColor(102, 102, 102);
            doc.setFont("montserrat", "normal");
            doc.setFontSize(12);

            doc.text(`Issued To: ${name}`, 20, 60);
            doc.text(`Age: ${document.getElementById('age').value}`, 20, 68);
            doc.text(`Nationality: ${document.getElementById('nationality').value}`, 20, 76);
            doc.text(`Contact: ${document.getElementById('contact').value}`, 20, 84);
            doc.text(`Email: ${document.getElementById('email').value}`, 20, 92);

            doc.setDrawColor(229, 229, 229);
            doc.line(20, 102, 190, 102);

            // Reservation Specifications Meta Table
            doc.setFontSize(14);
            doc.setTextColor(212, 175, 55);
            doc.text("Reservation Summary", 20, 115);

            doc.setFontSize(12);
            doc.setTextColor(102, 102, 102);
            doc.text(`Selected Tier: ${room}`, 20, 127);
            doc.text(`Check-In: ${checkIn}`, 20, 135);
            doc.text(`Check-Out: ${checkOut}`, 20, 143);
            doc.text(`Duration: ${totalNightsText}`, 20, 151);

            doc.setDrawColor(212, 175, 55);
            doc.line(20, 161, 190, 161);

            // Grand Total Pricing Breakdown display banner metric
            doc.setFontSize(16);
            doc.setTextColor(51, 51, 51);
            doc.text(`Grand Total Amount Due: ${totalCostText}`, 20, 175);

            doc.setFontSize(10);
            doc.setTextColor(153, 153, 153);
            doc.text("Thank you for choosing luxury. Please present this voucher upon check-in.", 20, 200);

            // Save system document output trigger file
            doc.save(`Dharryl_Hotel_Voucher_${name.replace(/\s+/g, '_')}.pdf`);
